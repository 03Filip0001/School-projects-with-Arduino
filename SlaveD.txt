#include <SoftwareSerial.h>

SoftwareSerial slave(2,3); // RX, TX

#define LED 5

void setup() {
  pinMode(LED , OUTPUT);
  
  Serial.begin(9600);

  slave.begin(9600);
}

void loop() {
  if(slave.available())
  {
    if(slave.read() == 0xFF)
    {
      digitalWrite(LED, HIGH);
      slave.print("Upalili smo");
      slave.flush();
    }
    else
    {
      digitalWrite(LED, LOW);
      slave.print("Ugaseni smo");
      slave.flush();
    }
  }

}
